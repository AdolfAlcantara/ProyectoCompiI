single_input -> stmt_list

stmt_list -> stmt stmt_list_p
stmt_list_p -> stmt stmt_list_p
stmt_list_p -> ''

stmt -> NewLine stmt_p
stmt_p -> assign
stmt_p -> print 
stmt_p -> '' 

print -> KwPrint String print_a
print_a -> , expr print_p
print_a -> ''
print_p -> , String , expr print_p
print_p -> ''

assign -> Ident = expr

expr -> term expr_p
expr_p -> + term expr_p
expr_p -> - term expr_p
expr_p -> ''

term -> prod term_p
term_p -> * prod term_p
term_p -> / prod term_p
term_p -> '' 

prod -> factor prod_p
prod_p -> % factor prod_p
prod_p -> Exp factor prod_p
prod_p -> ''

factor -> Ident
factor -> Number
factor -> ( expr )
factor -> func

func -> KwInput ( String )
